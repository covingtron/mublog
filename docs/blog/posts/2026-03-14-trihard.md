<script>
(function() {
  const container = document.getElementById('grid-container');
  const rowsInput = document.getElementById('grid-rows');
  const colsInput = document.getElementById('grid-cols');
  const colorB = document.getElementById('color-bg');
  const colorF = document.getElementById('color-fg');
  const colorA = document.getElementById('color-ac');
  const app = document.getElementById('trihard-app');

  let gridData = {}; // address -> "WNSE" (e.g., "BBBB")

  function getAddress(r, c) {
    const rowChar = String.fromCharCode(97 + r);
    const colStr = String(c + 1).padStart(2, '0');
    return rowChar + colStr;
  }

  function updateColors() {
    app.style.setProperty('--color-b', colorB.value);
    app.style.setProperty('--color-f', colorF.value);
    app.style.setProperty('--color-a', colorA.value);
  }

  function render() {
    const rows = Math.min(26, Math.max(1, parseInt(rowsInput.value) || 1));
    const cols = Math.min(99, Math.max(1, parseInt(colsInput.value) || 1));

    container.style.gridTemplateColumns = `repeat(${cols}, 40px)`;

    // Clear existing DOM but keep gridData
    container.innerHTML = '';

    for (let r = 0; r < rows; r++) {
      for (let c = 0; c < cols; c++) {
        const addr = getAddress(r, c);
        if (!gridData[addr]) {
          gridData[addr] = "BBBB";
        }

        const square = document.createElement('div');
        square.className = 'square';
        square.title = addr;

        const positions = ['W', 'N', 'S', 'E'];
        positions.forEach((pos, idx) => {
          const tri = document.createElement('div');
          const currentState = () => gridData[addr][idx];

          tri.className = `triangle ${pos} state-${currentState()}`;

          tri.onclick = (e) => {
            e.stopPropagation();
            const state = currentState();
            const nextState = state === 'B' ? 'F' : (state === 'F' ? 'A' : 'B');

            const dataArr = gridData[addr].split('');
            dataArr[idx] = nextState;
            gridData[addr] = dataArr.join('');

            tri.className = `triangle ${pos} state-${nextState}`;
            console.log(`Updated ${addr} position ${pos} to ${nextState}. Square state: ${gridData[addr]}`);
          };
          square.appendChild(tri);
        });
        container.appendChild(square);
      }
    }
  }

  rowsInput.oninput = render;
  colsInput.oninput = render;
  colorB.oninput = updateColors;
  colorF.oninput = updateColors;
  colorA.oninput = updateColors;

  // Initialize
  updateColors();
  render();
})();
</script>
